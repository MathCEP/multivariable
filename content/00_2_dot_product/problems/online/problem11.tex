\begin{problem}
Compute the projection of the vector $\vec{v}=(1,2,3)$ onto the vector $\vec{w} = (-3,2,-1)$.
\[
\textrm{proj}_{\vec{w}}(\vec{v}) = \answer{(0,0,0)}
\]
\begin{problem}
Why does your answer make sense?
\begin{multipleChoice}
\choice{The vectors are parallel.}
\choice[correct]{The vectors are perpendicular.}
\choice{They are the same length.}
\end{multipleChoice}
\end{problem}
\end{problem}
