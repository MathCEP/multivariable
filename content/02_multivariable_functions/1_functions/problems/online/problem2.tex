\begin{problem}
Let $f:\mathbb{R}^3\rightarrow\mathbb{R}^3$ be the function defined by
\[
f(\vec{x}) = 3\vec{x} + \textbf{i} - 2\textbf{j}.
\]

Find the component functions of $f$ in terms of $x$, $y$, and $z$.
\begin{align*}
f_1(x,y,z) &= \answer{3x+1}\\
f_2(x,y,z) &= \answer{3y-2}\\
f_3(x,y,z) &= \answer{3z}
\end{align*}
\end{problem}
