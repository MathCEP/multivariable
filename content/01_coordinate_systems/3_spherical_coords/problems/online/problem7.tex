$(\rho,\theta,\phi) = (\pi,\pi,\pi)$
\[
(x,y,z) = \answer{(0, 0, -\pi)}
\]
$(\rho,\theta,\phi) = (3,\pi/2,\pi/4)$
\[
(x,y,z) = \answer{(0, 3\sqrt{2}/2, 3\sqrt{2}/2)}
\]
$(\rho,\theta,\phi) = (2,7\pi/6,3\pi/4)$
\[
(x,y,z) = \answer{(-\sqrt{6}/2, -\sqrt{2}/2, -\sqrt{2})}
\]

\end{problem}
