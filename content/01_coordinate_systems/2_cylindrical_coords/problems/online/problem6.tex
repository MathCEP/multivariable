\begin{problem}
Find the Cartesian coordinates of each point, which is given in cylindrical coordinates.
$(r,\theta,z) = (1,1,1)$
\[
(x,y,z) = \answer{(\cos(1), \sin(1), 1)}
\]
$(r,\theta,z) = (\pi,\pi,\pi)$
\[
(x,y,z) = \answer{(-\pi, 0, \pi)}
\]
$(r,\theta,z) = (2,4\pi/3,-2)$
\[
(x,y,z) = \answer{(-\sqrt{3}, -1, -2)}
\]
\end{problem}
